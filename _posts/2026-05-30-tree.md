## 迭代

迭代相对复杂一些，但是也存在一个相对固定的模板，变化的是**记录答案的位置**。需要特别注意的是**后序遍历**，由于遍历完右节点需要知道根节点是什么，所以需要用`pre`记录根节点。

### 前序遍历

```c++
vector<int> preorder(TreeNode* root) {
	vector<int> ans;
	if (root == nullptr) {
		return ans;
	}
	stack<TreeNode*> stk;
	while (root != nullptr || !stk.empty()) {
		while (root != nullptr) {
			ans.push_back(root->val); // here
			stk.push(root);
			root = root->left;
		}
		root = stk.top();
		stk.pop();
		root = root->right;
	}
	return ans;
}
```

### 后序遍历

这里需要额外记录上一个节点，并且判断当前根节点是否需要重复入栈。

```c++
vector<int> postorder(TreeNode* root) {
	vector<int> ans;
	if (root == nullptr) {
		return ans;
	}
	stack<TreeNode*> stk;
	TreeNode* pre = nullptr; // here
	while (root != nullptr || !stk.empty()) {
		while (root != nullptr) {
			stk.push(root);
			root = root->left;
		}
		root = stk.top();
		stk.pop();

			root = nullptr;

		}
```

## 广度优先搜索

广度优先搜索是树和图的一种遍历方式，使用队列实现。这里用到图论中的BFS做层序遍历。

### 层序遍历

拓展每个节点的左右子节点

```c++
#include <iostream>
#include <vector>
#include <queue>
using namespace std;

struct TreeNode {
	int val;
	TreeNode* left;
	TreeNode* right;
	TreeNode() : val(0), left(nullptr), right(nullptr) {}
	TreeNode(int val) : val(val), left(nullptr), right(nullptr) {}
	TreeNode(int val, TreeNode* left, TreeNode* right) : val(val), left(left), right(right) {}
};

vector<vector<int>> levelorder(TreeNode* root) {
	vector<vector<int>> ans;
	if (root == nullptr) {
		return ans;
	}
	queue<TreeNode*> q;
	q.push(root);
	while (!q.empty()) {
		int n = q.size();
		vector<int> level;
		for (int i = 0; i < n; ++i) {
			root = q.front();
			q.pop();
			if (root->left != nullptr) {
				q.push(root->left);
			}
			if (root->right != nullptr) {
				q.push(root->right);
			}
			level.push_back(root->val);
		}
		ans.push_back(level);
	}
	return ans;
}


int main() {
	TreeNode* test1 = new TreeNode(1, nullptr, new TreeNode(2, new TreeNode(3), nullptr));
	TreeNode* test2 = new TreeNode(1, new TreeNode(2, new TreeNode(4), new TreeNode(5, new TreeNode(6), new TreeNode(7))), new TreeNode(3, nullptr, new TreeNode(8, new TreeNode(9), nullptr)));
	TreeNode* test3 = nullptr;
	TreeNode* test4 = new TreeNode(1);
	TreeNode* test5 = new TreeNode(3, new TreeNode(9), new TreeNode(20, new TreeNode(15), new TreeNode(7)));
	//vector<vector<int>> ans = levelorder(test1);
	//vector<vector<int>> ans = levelorder(test2);
	//vector<vector<int>> ans = levelorder(test3);
	//vector<vector<int>> ans = levelorder(test4);
	vector<vector<int>> ans = levelorder(test5);
	for (auto level : ans) {
		for (int num : level) {
```

## Morris

有一种巧妙的方法可以在线性时间内，只占用常数空间来实现前序遍历。这种方法由 J. H. Morris 在 1979 年的论文「[Traversing Binary Trees Simply and Cheaply](https://www.sciencedirect.com/science/article/abs/pii/0020019079900681?via%3Dihub){:target="_blank"}」中首次提出，因此被称为 Morris 遍历。Morris 遍历的核心思想是利用树的大量空闲指针，实现空间开销的极限缩减。

**总体代码依旧没有太大变化，变化的是加入答案数组的位置。**


### 前序遍历

假设当前遍历到的节点为`x`：

1. 如果`x`无左孩子，先将`x`的值加入答案数组，再访问`x`的右孩子，即`x=x.right`。
2. 如果`x`有左孩子，则找到`x`左子树上最右的节点（即左子树中序遍历的最后一个节点，`x`在中序遍历中的前驱节点），我们记为`predecessor`。根据`predecessor`的右孩子是否为空，进行如下操作。
  * 如果`predecessor`的右孩子为空，则将其右孩子指向`x`，**将`x`的值加入答案数组**，然后访问`x`的左孩子，即`x=x.left`。
  * 如果`predecessor`的右孩子不为空，则此时其右孩子指向`x`，说明我们已经遍历完`x`的左子树，我们将`predecessor`的右孩子置空，然后访问`x`的右孩子，即`x=x.right`。
3. 重复上述操作，直至访问完整棵树。

```c++
vector<int> preorder(TreeNode* root) {
	vector<int> ans;
	if (root == nullptr) {
		return ans;
	}
	TreeNode* pre = nullptr;
	while (root != nullptr) {
		if (root->left != nullptr) {
			// go to the right-most node
			pre = root->left;
			while (pre->right != nullptr && pre->right != root) {
				pre = pre->right;
			}
			// pre's right pointer to root
			if (pre->right == nullptr) {
				ans.push_back(root->val); // here
				pre->right = root;
				root = root->left;
			}
			// left subtree over go to right
			else {
				pre->right = nullptr;
				root = root->right;
			}
		}
		// no left subtree go to right
		else {
			ans.push_back(root->val);
			root = root->right;
		}
	}
	return ans;
}
```

### 后序遍历

假设当前遍历到的节点为`x`：

1. 如果`x`无左孩子，先将`x`的值加入答案数组，再访问`x`的右孩子，即`x=x.right`。
2. 如果`x`有左孩子，则找到`x`左子树上最右的节点（即左子树中序遍历的最后一个节点，`x`在中序遍历中的前驱节点），我们记为`predecessor`。根据`predecessor`的右孩子是否为空，进行如下操作。
  * 如果`predecessor`的右孩子为空，则将其右孩子指向`x`，然后访问`x`的左孩子，即`x=x.left`。
  * 如果`predecessor`的右孩子不为空，则此时其右孩子指向`x`，说明我们已经遍历完`x`的左子树，我们将`predecessor`的右孩子置空，**倒序输出从`x`的左子节点到该前驱节点这条路径上的所有节点**，然后访问`x`的右孩子，即`x=x.right`。
3. 重复上述操作，直至访问完整棵树。

需要用到一个辅助函数实现倒序输出：

```c++
void helper(vector<int>& vec, TreeNode* node) {
	int cnt = 0;
	while (node != nullptr) {
		++cnt;
		vec.push_back(node->val);
		node = node->right;
	}
	reverse(vec.end() - cnt, vec.end());
}
```

注意：必须要有dummy节点

```c++
}

vector<int> postorder(TreeNode* root) {
	vector<int> ans;
	if (root == nullptr) {
		return ans;
	}
	TreeNode* cur = new TreeNode(0, root, nullptr); // here
	TreeNode* pre = nullptr;
	while (cur != nullptr) {
		if (cur->left != nullptr) {
			// go to the right-most node
			pre = cur->left;
			while (pre->right != nullptr && pre->right != cur) {
				pre = pre->right;
			}
			// pre's right pointer to cur
			if (pre->right == nullptr) {
				pre->right = cur;
				cur = cur->left;
			}
			// left subtree over go to right
			else {
				pre->right = nullptr;
				helper(ans, cur->left); // here
				cur = cur->right;
			}
		}
		// no left subtree go to right
		else {
			cur = cur->right;
		}
	}
	return ans;
}
```

## 颜色标记法

一种解决迭代太复杂的简单写法，只需要调换一下顺序。注意，压栈顺序和遍历顺序是反的。

* 使用颜色标记节点的状态，新节点为白色，已访问的节点为灰色。
* 如果遇到的节点为白色，则将其标记为灰色，然后将其右子节点、自身、左子节点依次入栈。
* 如果遇到的节点为灰色，则将节点的值输出。

这种方法也叫“空间换智商”。

```c++
#include <iostream>
#include <vector>
#include <stack>
using namespace std;

struct TreeNode {
	int val;
	TreeNode* left;
	TreeNode* right;
	TreeNode() : val(0), left(nullptr), right(nullptr) {}
	TreeNode(int val) : val(val), left(nullptr), right(nullptr) {}
	TreeNode(int val, TreeNode* left, TreeNode* right) : val(val), left(left), right(right) {}
};

const int WHITE = 0, GRAY = 1;

vector<int> inorder(TreeNode* root) {
	vector<int> ans;
	if (root == nullptr) {
		return ans;
	}
	stack<pair<int, TreeNode*>> stk;
	stk.push({ WHITE, root });
	while (!stk.empty()) {
		auto [color, root] = stk.top();
		stk.pop();
		if (root == nullptr) {
			continue;
		}
		if (color == WHITE) { // here
			stk.push({ WHITE, root->right });
			stk.push({ GRAY, root });
			stk.push({ WHITE, root->left });
		}
		else {
			ans.push_back(root->val);
		}
	}
	return ans;
}

vector<int> preorder(TreeNode* root) {
	vector<int> ans;

	}

			stk.push({ GRAY, root });
		}

			ans.push_back(root->val);
		}
	}

}

		}
		else {

		}
	}
	return ans;
}
```
