Trie，又称前缀树或字典树，是一棵有根树，其每个节点包含以下字段：

* 指向子节点的指针数组`children`。对于本题而言，数组长度为26，即小写英文字母的数量。此时`children[0]`对应小写字母`a`，`children[1]`对应小写字母`b`，...，`children[25]`对应小写字母`z`。
* 布尔字段`isEnd`，表示该节点是否为字符串的结尾。

## 构造Trie树

Trie树有两个字段：

1. `bool isEnd`：表示该节点是否为字符串的结尾。
2. `vector<Trie*> children`：指向子节点的指针数组。

由此得到其构造函数：

```c++
class Trie {
private:
    bool isEnd;
    vector<Trie*> children;
public:
    Trie(): isEnd(false), children(26) {}
    void insert(string word) {}
    bool search(string word) {}
    bool startsWith(string prefix) {}
};
```


## 三种操作

1. 插入：将字符串插入到Trie树中。
2. 查询：判断字符串是否在Trie树中。
3. 匹配：判断字符串是否在Trie树中，并且该字符串是某个字符串的子串。

### void insert(string word)

1. 获取当前结点指针。
2. 对每个字符
  * 如果孩子空，`new Trie()`
  * 获取孩子指针
3. 最后一个字符设置`isEnd`为`true`

```c++
void insert(string word) {
  Trie* node = this;
  for (char c : word) {
    if (node->children[c - 'a'] == nullptr) {
      node->children[c - 'a'] = new Trie();
    }
    node = node->children[c - 'a'];
  }
  node->isEnd = true;
}
```

### bool search(string word)

1. 获取当前结点指针。
2. 对每个字符
  * 获取孩子指针
  * 如果孩子空，返回`false`
3. 返回最后一个字符的`isEnd`

```c++
bool search(string word) {
  Trie* node = this;
  for (char c : word) {
    node = node->children[c - 'a'];
    if (node == nullptr) {
      return false;
    }
  }
  return node->isEnd;
}
```

### bool startsWith(string prefix)

1. 获取当前结点指针。
2. 对每个字符
  * 获取孩子指针
  * 如果孩子空，返回`false`
3. 返回`true`


```c++
bool startsWith(string prefix) {
  Trie* node = this;
  for (char c : prefix) {
    node = node->children[c - 'a'];
    if (node == nullptr) {
      return false;
    }
  }
  return true;
}
```


## 完整代码

```c++
#include <iostream>
#include <vector>
#include <string>
using namespace std;

class Trie {
private:
	bool isEnd;
	vector<Trie*> children;

public:
	Trie() : isEnd(false), children(26) {};

	void insert(string word) {
		Trie* node = this;
		for (char c : word) {
			if (node->children[c - 'a'] == nullptr) {
				node->children[c - 'a'] = new Trie();
			}
			node = node->children[c - 'a'];
		}
		node->isEnd = true;
	}

	bool search(string word) {
		Trie* node = this;
		for (char c : word) {
			node = node->children[c - 'a'];
			if (node == nullptr) {
				return false;
			}
		}
		return node->isEnd;
	}

	bool startsWith(string prefix) {
		Trie* node = this;
		for (char c : prefix) {
			node = node->children[c - 'a'];
			if (node == nullptr) {
				return false;
			}
		}
		return true;
	}
};



int main() {
	Trie* test = new Trie();
	test->insert("apple");
	cout << test->search("apple");
	cout << test->search("app");
	cout << test->startsWith("app");
	test->insert("app");
	cout << test->search("app");
	return 0;
}
```
